```typescript
import { useEffect } from 'react'
import { useNavigate } from 'react-router-dom'
import { useState } from 'react'
import { useWizard } from './WizardContext'

export const StepDateTime = () => {
  const navigate = useNavigate()
  const { setDateTime } = useWizard()
  const [timeError, setTimeError] = useState<string | null>(null)

  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault()
    const fd = new FormData(e.currentTarget)
    const date = fd.get('date') as string
    const heure_debut = fd.get('heure_debut') as string
    const heure_fin = fd.get('heure_fin') as string
    const nombre_personnes = parseInt(fd.get('nombre_personnes') as string, 10)

    if (heure_fin <= heure_debut) {
      setTimeError("L'heure de fin doit être après l'heure de début.")
      return
    }

    setTimeError(null)
    setDateTime({ date, heure_debut, heure_fin, nombre_personnes })
    navigate('/reservations/table')
  }

  return (
    <div className="min-h-[60vh] flex items-center justify-center p-6">
      <div className="w-full max-w-md bg-surface rounded-2xl border border-white/5 p-8">
        <h2 className="text-xl font-semibold text-white mb-6 tracking-tight">
          Étape 1 — Date et horaires
        </h2>
        <form onSubmit={handleSubmit} aria-label="Formulaire de réservation" className="flex flex-col gap-5">
          <label className="flex flex-col gap-1.5 text-sm text-foreground-muted">
            Date
            <input
              type="date"
              name="date"
              required
              min={new Date().toISOString().split('T')[0]}
              className="bg-surface-elevated border border-white/10 rounded-lg px-4 py-2.5 text-white focus:outline-none focus:border-teal"
              style={{ colorScheme: 'dark' }}
            />
          </label>
          <label className="flex flex-col gap-1.5 text-sm text-foreground-muted">
            Heure d'arrivée
            <input
              type="time"
              name="heure_debut"
              required
              className="bg-surface-elevated border border-white/10 rounded-lg px-4 py-2.5 text-white focus:outline-none focus:border-teal"
              style={{ colorScheme: 'dark' }}
            />
          </label>
          <label className="flex flex-col gap-1.5 text-sm text-foreground-muted">
            Heure de départ
            <input
              type="time"
              name="heure_fin"
              required
              className="bg-surface-elevated border border-white/10 rounded-lg px-4 py-2.5 text-white focus:outline-none focus:border-teal"
              style={{ colorScheme: 'dark' }}
            />
          </label>
          <label className="flex flex-col gap-1.5 text-sm text-foreground-muted">
            Nombre de personnes
            <input
              type="number"
              name="nombre_personnes"
              required
              min={1}
              max={20}
              defaultValue={1}
              className="bg-surface-elevated border border-white/10 rounded-lg px-4 py-2.5 text-white focus:outline-none focus:border-teal"
            />
          </label>
          {timeError && (
            <p role="alert" className="text-sm text-error">{timeError}</p>
          )}
          <button
            type="submit"
            className="w-full bg-teal text-white rounded-lg py-3 font-semibold text-sm active:scale-[0.97]"
            style={{ transition: 'transform 160ms ease-out, opacity 180ms ease-out' }}
          >
            Choisir une table →
          </button>
        </form>
      </div>
    </div>
  )
}
```

**File 3: app/frontend/portail/src/pages/Reservations/StepTableSelect.tsx**

```typescript
import { useEffect, useState } from 'react'
import { useNavigate } from 'react-router-dom'
import { TableMap } from '@shared/components/map/TableMap'
import type { Table } from '@shared/types/tables'
import { useWizard } from './WizardContext'
import { fetchAvailableTables } from '../../api/reservations'

export const StepTableSelect = () => {
  const navigate = useNavigate()
  const { state, setTable } = useWizard()
  const [availableTables, setAvailableTables] = useState<Table[]>([])
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    if (!state.date) {
      navigate('/reservations/new', { replace: true })
      return
    }
    const load = async () => {
      try {
        const tables = await fetchAvailableTables({
          date: state.date,
          heure_debut: state.heure_debut,
          heure_fin: state.heure_fin,
          nombre_personnes: state.nombre_personnes,
        })
        setAvailableTables(tables)
      } catch (err) {
        console.error('Failed to fetch available tables', err)
        setError('Impossible de charger les tables disponibles. Veuillez réessayer.')
      } finally {
        setLoading(false)
      }
    }
    void load()
  }, [state.date, state.heure_debut, state.heure_fin, state.nombre_personnes, navigate])

  const handleTableClick = (table: Table) => {
    setTable(table)
    navigate('/reservations/confirm')
  }

  if (loading) {
    return (
      <div className="flex items-center justify-center min-h-[60vh]">
        <p className="text-foreground-muted text-sm animate-pulse">Chargement des tables disponibles…</p>
      </div>
    )
  }

  if (error) {
    return (
      <div className="flex items-center justify-center min-h-[60vh]">
        <p role="alert" className="text-error text-sm">{error}</p>
      </div>
    )
  }

  return (
    <div className="min-h-[60vh] flex flex-col p-6 gap-6">
      <div className="flex items-center justify-between">
        <h2 className="text-xl font-semibold text-white tracking-tight">
          Étape 2 — Choisissez votre table
        </h2>
        <button
          onClick={() => navigate('/reservations/new')}
          className="text-[11px] text-foreground-muted hover:text-white uppercase tracking-[0.15em] font-bold"
          style={{ transition: 'opacity 180ms ease-out' }}
        >
          ← Retour
        </button>
      </div>
      {availableTables.length === 0 ? (
        <p className="text-foreground-muted text-sm text-center mt-8">
          Aucune table disponible pour ce créneau. Modifiez vos horaires.
        </p>
      ) : (
        <TableMap
          tables={availableTables}
          onTableClick={handleTableClick}
          isEditMode={false}
        />
      )}
    </div>
  )
}
```

Do NOT pass `onTablePositionChange` — it is only used in edit mode. Do NOT modify `TableMap.tsx` or `TableItem.tsx`.

```typescript
import { useEffect, useState } from 'react'
import { useNavigate } from 'react-router-dom'
import { useWizard } from './WizardContext'
import { createReservation } from '../../api/reservations'

export const StepConfirm = () => {
  const navigate = useNavigate()
  const { state, reset } = useWizard()
  const [loading, setLoading] = useState(false)
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    if (!state.date || !state.selectedTable) {
      navigate('/reservations/new', { replace: true })
    }
  }, [state.date, state.selectedTable, navigate])

  const handleConfirm = async () => {
    setLoading(true)
    setError(null)
    try {
      await createReservation(state)
      reset()
      navigate('/reservations/new')
    } catch (err) {
      console.error('Reservation creation failed', err)
      setError('La réservation a échoué. Veuillez réessayer.')
    } finally {
      setLoading(false)
    }
  }

  if (!state.selectedTable) return null

  return (
    <div className="min-h-[60vh] flex items-center justify-center p-6">
      <div className="w-full max-w-md bg-surface rounded-2xl border border-white/5 p-8 flex flex-col gap-6">
        <h2 className="text-xl font-semibold text-white tracking-tight">
          Étape 3 — Confirmation
        </h2>
        <dl className="flex flex-col gap-3 text-sm">
          <div className="flex justify-between">
            <dt className="text-foreground-muted">Date</dt>
            <dd className="text-white font-medium">{state.date}</dd>
          </div>
          <div className="flex justify-between">
            <dt className="text-foreground-muted">Arrivée</dt>
            <dd className="text-white font-medium">{state.heure_debut}</dd>
          </div>
          <div className="flex justify-between">
            <dt className="text-foreground-muted">Départ</dt>
            <dd className="text-white font-medium">{state.heure_fin}</dd>
          </div>
          <div className="flex justify-between">
            <dt className="text-foreground-muted">Personnes</dt>
            <dd className="text-white font-medium">{state.nombre_personnes}</dd>
          </div>
          <div className="flex justify-between">
            <dt className="text-foreground-muted">Table</dt>
            <dd className="text-white font-medium">Table n°{state.selectedTable.numero}</dd>
          </div>
        </dl>
        {error && (
          <p role="alert" className="text-sm text-error">{error}</p>
        )}
        <div className="flex gap-3">
          <button
            onClick={() => navigate('/reservations/table')}
            className="flex-1 border border-white/10 text-foreground-muted rounded-lg py-3 text-sm font-medium active:scale-[0.97]"
            style={{ transition: 'transform 160ms ease-out, opacity 180ms ease-out' }}
          >
            ← Modifier
          </button>
          <button
            onClick={handleConfirm}
            disabled={loading}
            className="flex-1 bg-teal text-white rounded-lg py-3 text-sm font-semibold active:scale-[0.97] disabled:opacity-50"
            style={{ transition: 'transform 160ms ease-out, opacity 180ms ease-out' }}
          >
            {loading ? 'Confirmation…' : 'Confirmer la réservation'}
          </button>
        </div>
      </div>
    </div>
  )
}
```
