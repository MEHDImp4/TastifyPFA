# Minimal valid MPEG audio frame: 32-byte ID3v2.3 header + ~417-byte MPEG-1 Layer III frame (44.1kHz, 128kbps, mono, silence)
# The simplest portable approach: download an industry-standard 1-second silent royalty-free MP3 OR
# emit a minimal ID3 header + a single silent MPEG-1 Layer III frame.
$bytes = [byte[]]@(
  # ID3v2.3 header — "ID3", version 3.0, flags 0x00, size 0 (synchsafe)
  0x49,0x44,0x33,0x03,0x00,0x00,0x00,0x00,0x00,0x00,

  0xFF,0xFB,0x90,0x00

$frame = New-Object byte[] 417
